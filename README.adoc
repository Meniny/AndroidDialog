== 🔫 Usage

[source, swift]
----
import AndroidDialog
----

[source, swift]
----
let dialog = AndroidDialog.init()
dialog.setTitle("Alert")
dialog.setMessage("This is a message!!! O(∩_∩)O~")
dialog.setPositiveButton("Done") { (d) in
  d.hide()
}
dialog.coverType = .blurred(.dark)
dialog.show(to: self.view)
----
